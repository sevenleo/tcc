Com a preposição com
+ Artigos definidos
com + um = cum

Com a preposição de
+ Artigos definidos
de + o = do

de + a = da

de + os = dos

de + as = das



+ Artigos indefinidos
de + um = dum

de + uns = duns

de + uma = duma

de + umas = dumas



+ Pronomes pessoais
de + ele = dele

de + ela = dela

de + eles = deles

de + elas = delas



+ Pronomes demonstrativos
de + este = deste

de + esta = desta

de + estes = destes

de + estas = destas

de + isto = disto

de + esse = desse

de + essa = dessa

de + esses = desses

de + essas = dessas

de + isso = disso

de + aquele = daquele

de + aquela = daquela

de + aqueles = daqueles

de + aquelas = daquelas

de + aquilo = daquilo



+ Pronome indefinido
de + outro = doutro

de + outra = doutra

de + outros = doutros

de + outras = doutra



+ Advérbios
de + aqui = daqui

de + aí = daí

de + ali = dali

de + além = dalém
